package vacaciones.ws

import com.squareup.moshi.Moshi
import com.squareup.moshi.kotlin.reflect.KotlinJsonAdapterFactory
import retrofit2.Retrofit
import retrofit2.converter.moshi.MoshiConverterFactory

class CambioServiceFactory {

    companion object{
        fun getCambioService(): CambioService {
            val moshi = Moshi.Builder()
                .add( KotlinJsonAdapterFactory() )
                .build()
            val converterFactory = MoshiConverterFactory.create( moshi )
            val baseUrl = "https://www.mindicador.cl"
            val retrofit = Retrofit.Builder()
                .addConverterFactory( converterFactory )
                .baseUrl( baseUrl )
                .build()
            return retrofit.create( CambioService::class.java)
        }
    }
}

object CambioAPI {
    val service: CambioService by lazy {
        CambioServiceFactory.getCambioService()
    }
}
